<script>
	function toggle(){
		var ele=document.getElementById("image");
		if(ele.style.display=="block"){
			ele.style.display="none";
		}
		else{
			ele.style.display="block";
		}
	}
	function addlist(){
		var form=document.getElementById("pizza"); 
		ele=document.createElement("li");
		ele.innerHTML=form.variety.value;
		document.getElementById("pizzalist").appendChild(ele);	
		return true;
	}
</script>
